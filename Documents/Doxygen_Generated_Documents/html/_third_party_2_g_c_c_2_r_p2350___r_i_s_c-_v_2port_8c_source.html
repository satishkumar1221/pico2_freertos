189void vPortSetupTimerInterrupt( void )
190{
191 riscv_timer_set_fullspeed(true);
192 uint64_t ullCurrentTime = riscv_timer_get_mtime();
193 uxTimerIncrementsForOneTick = ( size_t ) ( ( clock_get_hz( clk_sys ) ) / ( configTICK_RATE_HZ ) ); /* Assumes increment won't go over 32-bits. */
194
195 pullMachineTimerCompareRegister = ( volatile uint64_t * ) &sio_hw->mtimecmp;
196
197 ullNextTime = ullCurrentTime + uxTimerIncrementsForOneTick;
198 *pullMachineTimerCompareRegister = ullNextTime;
199
200 /* Prepare the time to use after the next tick interrupt. */
201 ullNextTime += uxTimerIncrementsForOneTick;
202
203 extern void freertos_risc_v_mtimer_interrupt_handler(void);
204#if 1
205 irq_set_exclusive_handler(SIO_IRQ_MTIMECMP, freertos_risc_v_mtimer_interrupt_handler);
206 irq_set_enabled(SIO_IRQ_MTIMECMP, true);
207#else
208 riscv_set_csr(mie, RVCSR_MIE_MTIE_BITS);
209#endif
210}

486 void vPortLockInternalSpinUnlockWithWait( struct lock_core * pxLock,
487 uint32_t ulSave )
488 {
489 configASSERT( !portCHECK_IF_IN_ISR() );
490 configASSERT( pxLock->spin_lock );
491
492 if( !portIS_FREE_RTOS_CORE() )
493 {
494 spin_unlock( pxLock->spin_lock, ulSave );
495 __wfe();
496 }
497 else
498 {
499 /* The requirement (from the SDK) on this implementation is that this method
500 * should always wake up from a corresponding call to vPortLockInternalSpinUnlockWithNotify
501 * that happens after this method is called.
502 *
503 * The moment that we unlock the spin lock, we need to be sure that
504 * there is no way that we end up blocking in xEventGroupWaitBits,
505 * despite the fact that other tasks can now run, if the corresponding
506 * unlock has occurred.
507 *
508 * Previously the RP2xxx ports used to attempt to disable IRQs until the
509 * task actually (potentially) became blocked by hooking the IRQ re-enable
510 * when xEventGroupWaitBits completes (or switches tasks), but this
511 * was a broken hack, in that IRQs are re-enabled at other points during
512 * that call.
513 *
514 * This deferred IRQ enable is not actually needed, because all we
515 * care about is that:
516 *
517 * Even in the presence of other tasks acquiring then releasing
518 * the lock, between the interrupt_enable and the xEventGroupWaitBits,
519 * the corresponding bit will still be set.
520 *
521 * This is the case, even any intervening blocked lock (which
522 * clears the event bit) will need to unlock it before we proceed,
523 * which will set the event bit again.
524 *
525 * The multiplexing down of multiple spin lock numbers to fewer
526 * event bits does not cause a possible race condition,
527 * but it does mean that a task waiting for lock A can be
528 * blocked by a task B which owns another lock.
529 *
530 * This could be fixed by using an array of event groups, however
531 * since the SDK spin locks are generally intended for very short
532 * term usage anyway, and rarely nested except in exotic cases
533 * like video output, we'll leave it as one event group for now
534 */
535 spin_unlock( pxLock->spin_lock, ulSave);
536 xEventGroupWaitBits( xEventGroup, prvGetEventGroupBit( pxLock->spin_lock ),
537 pdTRUE, pdFALSE, portMAX_DELAY );
538 }
539 }

541 void vPortLockInternalSpinUnlockWithNotify( struct lock_core * pxLock,
542 uint32_t ulSave )
543 {
544 EventBits_t uxBits = prvGetEventGroupBit( pxLock->spin_lock );
545
546 if( portIS_FREE_RTOS_CORE() )
547 {
548 #if LIB_PICO_MULTICORE
549 /* signal an event in case a regular core is waiting */
550 __sev();
551 #endif
552 spin_unlock( pxLock->spin_lock, ulSave );
553
554 if( !portCHECK_IF_IN_ISR() )
555 {
556 xEventGroupSetBits( xEventGroup, uxBits );
557 }
558 else
559 {
560 BaseType_t xHigherPriorityTaskWoken = pdFALSE;
561 xEventGroupSetBitsFromISR( xEventGroup, uxBits, &xHigherPriorityTaskWoken );
562 portYIELD_FROM_ISR( xHigherPriorityTaskWoken );
563 }
564 }
565 else
566 {
567 __sev();
568 #if ( configNUMBER_OF_CORES == 1 )
569 if( pxCrossCoreSpinLock != pxLock->spin_lock )
570 {
571 spin_lock_unsafe_blocking( pxCrossCoreSpinLock );
572 uxCrossCoreEventBits |= uxBits;
573 spin_unlock_unsafe( pxCrossCoreSpinLock );
574 }
575 else
576 {
577 uxCrossCoreEventBits |= uxBits;
578 }
579
580 /* This causes doorbell irq on the other (FreeRTOS) core which will do the set the event bits */
581 if ( cDoorbellNum >= 0 )
582 {
583 multicore_doorbell_set_other_core(cDoorbellNum);
584 }

590 bool xPortLockInternalSpinUnlockWithBestEffortWaitOrTimeout( struct lock_core * pxLock,
591 uint32_t ulSave,
592 absolute_time_t uxUntil )
593 {
594 configASSERT( !portCHECK_IF_IN_ISR() );
595 configASSERT( pxLock->spin_lock );
596
597 /* note no need to check LIB_PICO_MULTICORE, as this is always returns true if that is not defined */
598 if( !portIS_FREE_RTOS_CORE() )
599 {
600 spin_unlock( pxLock->spin_lock, ulSave );
601 return best_effort_wfe_or_timeout( uxUntil );
602 }
603 else
604 {
605 configASSERT( portIS_FREE_RTOS_CORE() );
606
607 TickType_t uxTicksToWait = prvGetTicksToWaitBefore( uxUntil );
608
609 if( uxTicksToWait )
610 {
611 /* See comment in vPortLockInternalSpinUnlockWithWait for detail
612 * about possible race conditions */
613 spin_unlock( pxLock->spin_lock, ulSave );
614 xEventGroupWaitBits( xEventGroup,
615 prvGetEventGroupBit( pxLock->spin_lock ), pdTRUE,
616 pdFALSE, uxTicksToWait );
617 }
618 else
619 {
620 spin_unlock( pxLock->spin_lock, ulSave );
621 }
622
623 if( time_reached( uxUntil ) )
624 {
625 return true;
626 }
627 else
628 {
629 /* We do not want to hog the core */
630 portYIELD();
631 /* We aren't sure if we've reached the timeout yet; the caller will check */
632 return false;
633 }
634 }

58/* Only the standard core registers are stored by default. Any additional
59 * registers must be saved by the portasmSAVE_ADDITIONAL_REGISTERS and
60 * portasmRESTORE_ADDITIONAL_REGISTERS macros - which can be defined in a chip
61 * specific version of freertos_risc_v_chip_specific_extensions.h. See the
62 * notes at the top of portASM.S file. */
63#ifdef __riscv_32e
64 #define portCONTEXT_SIZE ( 15 * portWORD_SIZE )
65 #define portCRITICAL_NESTING_OFFSET 13
66 #define portMSTATUS_OFFSET 14
67#else
68 #define portCONTEXT_SIZE ( 31 * portWORD_SIZE )
69 #define portCRITICAL_NESTING_OFFSET 29
70 #define portMSTATUS_OFFSET 30
71#endif

96.macro portcontextGET_CURRENT_TCB_FOR_CORE core_reg, out_reg

109.macro portcontextSWITCH_TO_ISR_STACK reg_pISRStackTop, reg_ISRStackTop
110 la \reg_pISRStackTop, SIO_BASE
111 load_x \reg_pISRStackTop, 0 ( \reg_pISRStackTop )
112 la \reg_ISRStackTop, xISRStackTops
113 sh2add \reg_pISRStackTop, \reg_pISRStackTop, \reg_ISRStackTop
114 load_x \reg_ISRStackTop, 0 ( \reg_pISRStackTop )
115 beqz \reg_ISRStackTop, no_isr_stack\@
116 mv sp, \reg_ISRStackTop
117no_isr_stack\@:
118.endm

174portcontextSAVE_CONTEXT_INTERNAL

183 .macro portcontextSAVE_INTERRUPT_CONTEXT
184portcontextSAVE_CONTEXT_INTERNAL
185csrr a0, mcause
186csrr a1, mepc
187store_x a1, 0 ( sp ) /* Asynchronous interrupt so save unmodified exception return address. */
188portcontextSWITCH_TO_ISR_STACK s10, s11
189/*
190 * s10 is now pISRStackTop for the current core
191 * s11 is the old value of ISRStackTop
192 * we expect them to remain saved, and we overwrite the
193 * pISRStackTop with zero, so we don't overwrite again if nesting is enabled
194 */
195 sw zero, 0 (s10)
196 .endm

204/* Load mepc with the address of the instruction in the task to run next. */
205load_x t0, 0 ( sp )

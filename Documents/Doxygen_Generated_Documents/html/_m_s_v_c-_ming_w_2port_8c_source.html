42#define portMAX_INTERRUPTS ( ( uint32_t ) sizeof( uint32_t ) * 8UL ) /* The number of bits in an uint32_t. */
43#define portNO_CRITICAL_NESTING ( ( uint32_t ) 0 )

51/*
52 * Created as a high priority thread, this function uses a timer to simulate
53 * a tick interrupt being generated on an embedded target. In this Windows
54 * environment the timer does not achieve anything approaching real time
55 * performance though.

72/*
73 * Exiting a critical section will cause the calling task to block on yield
74 * event to wait for an interrupt to process if an interrupt was pended while
75 * inside the critical section. This variable protects against a recursive
76 * attempt to obtain pvInterruptEventMutex if a critical section is used inside
77 * an interrupt handler itself.

79volatile BaseType_t xInsideInterrupt = pdFALSE;

89/* The WIN32 simulator runs each task in a thread. The context switching is
90 * managed by the threads, so the task stack does not have to be managed directly,
91 * although the task stack is still used to hold an xThreadState structure this is
92 * the only thing it will ever hold. The structure indirectly maps the task handle
93 * to a thread handle. */
94typedef struct
95{
96 /* Handle of the thread that executes the task. */
97 void * pvThread;
98
99 /* Event used to make sure the thread does not execute past a yield point
100 * between the call to SuspendThread() to suspend the thread and the
101 * asynchronous SuspendThread() operation actually being performed. */
102 void * pvYieldEvent;
103} ThreadState_t;
104
105/* Simulated interrupts waiting to be processed. This is a bit mask where each
106 * bit represents one interrupt, so a maximum of 32 interrupts can be simulated. */
107static volatile uint32_t ulPendingInterrupts = 0UL;
108
109/* An event used to inform the simulated interrupt processing thread (a high
110 * priority thread that simulated interrupt processing) that an interrupt is
111 * pending. */
112static void * pvInterruptEvent = NULL;
113
114/* Mutex used to protect all the simulated interrupt variables that are accessed
115 * by multiple threads. */
116static void * pvInterruptEventMutex = NULL;
117
118/* The critical nesting count for the currently executing task. This is
119 * initialised to a non-zero value so interrupts do not become enabled during
120 * the initialisation phase. As each task has its own critical nesting value
121 * ulCriticalNesting will get set to zero when the first task runs. This
122 * initialisation is probably not critical in this simulated environment as the
123 * simulated interrupt handlers do not get created until the FreeRTOS scheduler is
124 * started anyway. */
125static volatile uint32_t ulCriticalNesting = 9999UL;
126
127/* Handlers for all the simulated software interrupts. The first two positions
128 * are used for the Yield and Tick interrupts so are handled slightly differently,
129 * all the other interrupts can be user defined. */
130static uint32_t (* ulIsrHandler[ portMAX_INTERRUPTS ])( void ) = { 0 };
131
132/* Pointer to the TCB of the currently executing task. */
133extern void * volatile pxCurrentTCB;
134
135/* Used to ensure nothing is processed during the startup sequence. */
136static BaseType_t xPortRunning = pdFALSE;
137

140static DWORD WINAPI prvSimulatedPeripheralTimer( LPVOID lpParameter )
141{
142 TickType_t xMinimumWindowsBlockTime;
143 TIMECAPS xTimeCaps;
144
145 /* Set the timer resolution to the maximum possible. */
146 if( timeGetDevCaps( &xTimeCaps, sizeof( xTimeCaps ) ) == MMSYSERR_NOERROR )
147 {
148 xMinimumWindowsBlockTime = ( TickType_t ) xTimeCaps.wPeriodMin;
149 timeBeginPeriod( xTimeCaps.wPeriodMin );
150
151 /* Register an exit handler so the timeBeginPeriod() function can be
152 * matched with a timeEndPeriod() when the application exits. */
153 SetConsoleCtrlHandler( prvEndProcess, TRUE );
154 }
155 else
156 {
157 xMinimumWindowsBlockTime = ( TickType_t ) 20;
158 }
159
160 /* Just to prevent compiler warnings. */
161 ( void ) lpParameter;
162
163 while( xPortRunning == pdTRUE )
164 {
165 /* Wait until the timer expires and we can access the simulated interrupt
166 * variables. *NOTE* this is not a 'real time' way of generating tick
167 * events as the next wake time should be relative to the previous wake
168 * time, not the time that Sleep() is called. It is done this way to
169 * prevent overruns in this very non real time simulated/emulated
170 * environment. */
171 if( portTICK_PERIOD_MS < xMinimumWindowsBlockTime )
172 {
173 Sleep( xMinimumWindowsBlockTime );
174 }
175 else
176 {
177 Sleep( portTICK_PERIOD_MS );
178 }
179
180 vPortGenerateSimulatedInterruptFromWindowsThread( portINTERRUPT_TICK );
181 }
182
183 return 0;
184}

187static BOOL WINAPI prvEndProcess( DWORD dwCtrlType )
188{
189 TIMECAPS xTimeCaps;
190
191 ( void ) dwCtrlType;
192
193 if( timeGetDevCaps( &xTimeCaps, sizeof( xTimeCaps ) ) == MMSYSERR_NOERROR )
194 {
195 /* Match the call to timeBeginPeriod( xTimeCaps.wPeriodMin ) made when
196 * the process started with a timeEndPeriod() as the process exits. */
197 timeEndPeriod( xTimeCaps.wPeriodMin );
198 }
199
200 return pdFALSE;
201}

204StackType_t * pxPortInitialiseStack( StackType_t * pxTopOfStack,
205 TaskFunction_t pxCode,
206 void * pvParameters )
207{
208 ThreadState_t * pxThreadState = NULL;
209 int8_t * pcTopOfStack = ( int8_t * ) pxTopOfStack;
210 const SIZE_T xStackSize = 1024; /* Set the size to a small number which will get rounded up to the minimum possible. */
211
212 /* In this simulated case a stack is not initialised, but instead a thread
213 * is created that will execute the task being created. The thread handles
214 * the context switching itself. The ThreadState_t object is placed onto
215 * the stack that was created for the task - so the stack buffer is still
216 * used, just not in the conventional way. It will not be used for anything
217 * other than holding this structure. */
218 pxThreadState = ( ThreadState_t * ) ( pcTopOfStack - sizeof( ThreadState_t ) );
219
220 /* Create the event used to prevent the thread from executing past its yield
221 * point if the SuspendThread() call that suspends the thread does not take
222 * effect immediately (it is an asynchronous call). */
223 pxThreadState->pvYieldEvent = CreateEvent( NULL, /* Default security attributes. */
224 FALSE, /* Auto reset. */
225 FALSE, /* Start not signalled. */
226 NULL ); /* No name. */
227
228
229#ifdef __GNUC__
230 /* GCC reports the warning for the cast operation from TaskFunction_t to LPTHREAD_START_ROUTINE. */
231 /* Disable this warning here by the #pragma option. */
232#pragma GCC diagnostic push
233#pragma GCC diagnostic ignored "-Wcast-function-type"
234#endif
235 /* Create the thread itself. */
236 pxThreadState->pvThread = CreateThread( NULL, xStackSize, ( LPTHREAD_START_ROUTINE ) pxCode, pvParameters, CREATE_SUSPENDED | STACK_SIZE_PARAM_IS_A_RESERVATION, NULL );
237#ifdef __GNUC__
238#pragma GCC diagnostic pop
239#endif
240
241 configASSERT( pxThreadState->pvThread ); /* See comment where TerminateThread() is called. */
242 SetThreadAffinityMask( pxThreadState->pvThread, 0x01 );
243 SetThreadPriorityBoost( pxThreadState->pvThread, TRUE );
244 SetThreadPriority( pxThreadState->pvThread, portTASK_THREAD_PRIORITY );
245
246 return ( StackType_t * ) pxThreadState;
247}

250BaseType_t xPortStartScheduler( void )
251{
252 void * pvHandle = NULL;
253 int32_t lSuccess;
254 ThreadState_t * pxThreadState = NULL;
255 SYSTEM_INFO xSystemInfo;
256
257 /* This port runs windows threads with extremely high priority. All the
258 * threads execute on the same core - to prevent locking up the host only start
259 * if the host has multiple cores. */
260 GetSystemInfo( &xSystemInfo );
261
262 if( xSystemInfo.dwNumberOfProcessors <= 1 )
263 {
264 printf( "This version of the FreeRTOS Windows port can only be used on multi-core hosts.\r\n" );
265 lSuccess = pdFAIL;
266 }
267 else
268 {
269 lSuccess = pdPASS;
270
271 /* The highest priority class is used to [try to] prevent other Windows
272 * activity interfering with FreeRTOS timing too much. */
273 if( SetPriorityClass( GetCurrentProcess(), REALTIME_PRIORITY_CLASS ) == 0 )
274 {
275 printf( "SetPriorityClass() failed\r\n" );
276 }
277
278 /* Install the interrupt handlers used by the scheduler itself. */
279 vPortSetInterruptHandler( portINTERRUPT_YIELD, prvProcessYieldInterrupt );
280 vPortSetInterruptHandler( portINTERRUPT_TICK, prvProcessTickInterrupt );
281
282 /* Create the events and mutexes that are used to synchronise all the
283 * threads. */
284 pvInterruptEventMutex = CreateMutex( NULL, FALSE, NULL );
285 pvInterruptEvent = CreateEvent( NULL, FALSE, FALSE, NULL );
286
287 if( ( pvInterruptEventMutex == NULL ) || ( pvInterruptEvent == NULL ) )
288 {
289 lSuccess = pdFAIL;
290 }
291
292 /* Set the priority of this thread such that it is above the priority of
293 * the threads that run tasks. This higher priority is required to ensure
294 * simulated interrupts take priority over tasks. */
295 pvHandle = GetCurrentThread();
296
297 if( pvHandle == NULL )
298 {
299 lSuccess = pdFAIL;
300 }
301 }
302
303 if( lSuccess == pdPASS )
304 {
305 if( SetThreadPriority( pvHandle, portSIMULATED_INTERRUPTS_THREAD_PRIORITY ) == 0 )
306 {
307 lSuccess = pdFAIL;
308 }
309
310 SetThreadPriorityBoost( pvHandle, TRUE );
311 SetThreadAffinityMask( pvHandle, 0x01 );
312 }
313
314 if( lSuccess == pdPASS )
315 {
316 /* Start the thread that simulates the timer peripheral to generate
317 * tick interrupts. The priority is set below that of the simulated
318 * interrupt handler so the interrupt event mutex is used for the
319 * handshake / overrun protection. */
320 pvHandle = CreateThread( NULL, 0, prvSimulatedPeripheralTimer, NULL, CREATE_SUSPENDED, NULL );
321
322 if( pvHandle != NULL )
323 {
324 SetThreadPriority( pvHandle, portSIMULATED_TIMER_THREAD_PRIORITY );
325 SetThreadPriorityBoost( pvHandle, TRUE );
326 SetThreadAffinityMask( pvHandle, 0x01 );
327 ResumeThread( pvHandle );
328 }
329
330 /* Start the highest priority task by obtaining its associated thread
331 * state structure, in which is stored the thread handle. */
332 pxThreadState = ( ThreadState_t * ) *( ( size_t * ) pxCurrentTCB );
333 ulCriticalNesting = portNO_CRITICAL_NESTING;
334
335 /* The scheduler is now running. */
336 xPortRunning = pdTRUE;
337
338 /* Start the first task. */
339 ResumeThread( pxThreadState->pvThread );
340
341 /* Handle all simulated interrupts - including yield requests and
342 * simulated ticks. */
343 prvProcessSimulatedInterrupts();
344 }
345
346 /* Would not expect to return from prvProcessSimulatedInterrupts(), so should
347 * not get here. */
348 return 0;
349}

370
371static void prvProcessSimulatedInterrupts( void )
372{
373 uint32_t ulSwitchRequired, i;
374 ThreadState_t * pxThreadState;
375 void * pvObjectList[ 2 ];
376 CONTEXT xContext;
377 DWORD xWinApiResult;
378 const DWORD xTimeoutMilliseconds = 1000;
379
380 /* Going to block on the mutex that ensured exclusive access to the simulated
381 * interrupt objects, and the event that signals that a simulated interrupt
382 * should be processed. */
383 pvObjectList[ 0 ] = pvInterruptEventMutex;
384 pvObjectList[ 1 ] = pvInterruptEvent;
385
386 /* Create a pending tick to ensure the first task is started as soon as
387 * this thread pends. */
388 ulPendingInterrupts |= ( 1 << portINTERRUPT_TICK );
389 SetEvent( pvInterruptEvent );
390
391 while( xPortRunning == pdTRUE )
392 {
393 xInsideInterrupt = pdFALSE;
394
395 /* Wait with timeout so that we can exit from this loop when
396 * the scheduler is stopped by calling vPortEndScheduler. */
397 xWinApiResult = WaitForMultipleObjects( sizeof( pvObjectList ) / sizeof( void * ), pvObjectList, TRUE, xTimeoutMilliseconds );
398
399 if( xWinApiResult != WAIT_TIMEOUT )
400 {
401 /* Cannot be in a critical section to get here. Tasks that exit a
402 * critical section will block on a yield mutex to wait for an interrupt to
403 * process if an interrupt was set pending while the task was inside the
404 * critical section. xInsideInterrupt prevents interrupts that contain
405 * critical sections from doing the same. */
406 xInsideInterrupt = pdTRUE;
407
408 /* Used to indicate whether the simulated interrupt processing has
409 * necessitated a context switch to another task/thread. */
410 ulSwitchRequired = pdFALSE;
411
412 /* For each interrupt we are interested in processing, each of which is
413 * represented by a bit in the 32bit ulPendingInterrupts variable. */
414 for( i = 0; i < portMAX_INTERRUPTS; i++ )
415 {
416 /* Is the simulated interrupt pending? */
417 if( ( ulPendingInterrupts & ( 1UL << i ) ) != 0 )
418 {
419 /* Is a handler installed? */
420 if( ulIsrHandler[ i ] != NULL )
421 {
422 /* Run the actual handler. Handlers return pdTRUE if they
423 * necessitate a context switch. */
424 if( ulIsrHandler[ i ]() != pdFALSE )
425 {
426 /* A bit mask is used purely to help debugging. */
427 ulSwitchRequired |= ( 1 << i );
428 }
429 }
430
431 /* Clear the interrupt pending bit. */
432 ulPendingInterrupts &= ~( 1UL << i );
433 }
434 }
435
436 if( ulSwitchRequired != pdFALSE )
437 {
438 /* Suspend the old thread. */
439 pxThreadState = ( ThreadState_t * ) *( ( size_t * ) pxCurrentTCB );
440 SuspendThread( pxThreadState->pvThread );
441
442 /* Ensure the thread is actually suspended by performing a
443 * synchronous operation that can only complete when the thread
444 * is actually suspended. The below code asks for dummy register
445 * data. Experimentation shows that these two lines don't appear
446 * to do anything now, but according to
447 * https://devblogs.microsoft.com/oldnewthing/20150205-00/?p=44743
448 * they do - so as they do not harm (slight run-time hit). */
449 xContext.ContextFlags = CONTEXT_INTEGER;
450 ( void ) GetThreadContext( pxThreadState->pvThread, &xContext );
451
452 /* Select the next task to run. */
453 vTaskSwitchContext();
454
455 /* Obtain the state of the task now selected to enter the
456 * Running state. */
457 pxThreadState = ( ThreadState_t * ) ( *( size_t * ) pxCurrentTCB );
458
459 /* pxThreadState->pvThread can be NULL if the task deleted
460 * itself - but a deleted task should never be resumed here. */
461 configASSERT( pxThreadState->pvThread != NULL );
462 ResumeThread( pxThreadState->pvThread );
463 }
464
465 /* If the thread that is about to be resumed stopped running
466 * because it yielded then it will wait on an event when it resumed
467 * (to ensure it does not continue running after the call to
468 * SuspendThread() above as SuspendThread() is asynchronous).
469 * Signal the event to ensure the thread can proceed now it is
470 * valid for it to do so. Signaling the event is benign in the case that
471 * the task was switched out asynchronously by an interrupt as the event
472 * is reset before the task blocks on it. */
473 pxThreadState = ( ThreadState_t * ) ( *( size_t * ) pxCurrentTCB );
474 SetEvent( pxThreadState->pvYieldEvent );
475 ReleaseMutex( pvInterruptEventMutex );
476 }
477 }
478}

481void vPortDeleteThread( void * pvTaskToDelete )
482{
483 ThreadState_t * pxThreadState;
484 uint32_t ulErrorCode;
485
486 /* Remove compiler warnings if configASSERT() is not defined. */
487 ( void ) ulErrorCode;
488
489 /* Find the handle of the thread being deleted. */
490 pxThreadState = ( ThreadState_t * ) ( *( size_t * ) pvTaskToDelete );
491
492 /* Check that the thread is still valid, it might have been closed by
493 * vPortCloseRunningThread() - which will be the case if the task associated
494 * with the thread originally deleted itself rather than being deleted by a
495 * different task. */
496 if( pxThreadState->pvThread != NULL )
497 {
498 WaitForSingleObject( pvInterruptEventMutex, INFINITE );
499
500 /* !!! This is not a nice way to terminate a thread, and will eventually
501 * result in resources being depleted if tasks frequently delete other
502 * tasks (rather than deleting themselves) as the task stacks will not be
503 * freed. */
504 ulErrorCode = TerminateThread( pxThreadState->pvThread, 0 );
505 configASSERT( ulErrorCode );
506
507 ulErrorCode = CloseHandle( pxThreadState->pvThread );
508 configASSERT( ulErrorCode );
509
510 ReleaseMutex( pvInterruptEventMutex );
511 }
512}
513/*-----------------------------------------------------------*/
514
515void vPortCloseRunningThread( void * pvTaskToDelete,
516 volatile BaseType_t * pxPendYield )
517{
518 ThreadState_t * pxThreadState;
519 void * pvThread;
520 uint32_t ulErrorCode;
521
522 /* Remove compiler warnings if configASSERT() is not defined. */
523 ( void ) ulErrorCode;
524
525 /* Find the handle of the thread being deleted. */
526 pxThreadState = ( ThreadState_t * ) ( *( size_t * ) pvTaskToDelete );
527 pvThread = pxThreadState->pvThread;
528
529 /* Raise the Windows priority of the thread to ensure the FreeRTOS scheduler
530 * does not run and swap it out before it is closed. If that were to happen
531 * the thread would never run again and effectively be a thread handle and
532 * memory leak. */
533 SetThreadPriority( pvThread, portDELETE_SELF_THREAD_PRIORITY );
534
535 /* This function will not return, therefore a yield is set as pending to
536 * ensure a context switch occurs away from this thread on the next tick. */
537 *pxPendYield = pdTRUE;
538
539 /* Mark the thread associated with this task as invalid so
540 * vPortDeleteThread() does not try to terminate it. */
541 pxThreadState->pvThread = NULL;
542
543 /* Close the thread. */
544 ulErrorCode = CloseHandle( pvThread );
545 configASSERT( ulErrorCode );
546
547 /* This is called from a critical section, which must be exited before the
548 * thread stops. */
549 taskEXIT_CRITICAL();
550 CloseHandle( pxThreadState->pvYieldEvent );
551 ExitThread( 0 );
552}

561void vPortGenerateSimulatedInterrupt( uint32_t ulInterruptNumber )
562{
563 ThreadState_t * pxThreadState = ( ThreadState_t * ) *( ( size_t * ) pxCurrentTCB );
564
565 configASSERT( xPortRunning );
566
567 if( ( ulInterruptNumber < portMAX_INTERRUPTS ) && ( pvInterruptEventMutex != NULL ) )
568 {
569 WaitForSingleObject( pvInterruptEventMutex, INFINITE );
570 ulPendingInterrupts |= ( 1 << ulInterruptNumber );
571
572 /* The simulated interrupt is now held pending, but don't actually
573 * process it yet if this call is within a critical section. It is
574 * possible for this to be in a critical section as calls to wait for
575 * mutexes are accumulative. If in a critical section then the event
576 * will get set when the critical section nesting count is wound back
577 * down to zero. */
578 if( ulCriticalNesting == portNO_CRITICAL_NESTING )
579 {
580 SetEvent( pvInterruptEvent );
581
582 /* Going to wait for an event - make sure the event is not already
583 * signaled. */
584 ResetEvent( pxThreadState->pvYieldEvent );
585 }
586
587 ReleaseMutex( pvInterruptEventMutex );
588
589 if( ulCriticalNesting == portNO_CRITICAL_NESTING )
590 {
591 /* An interrupt was pended so ensure to block to allow it to
592 * execute. In most cases the (simulated) interrupt will have
593 * executed before the next line is reached - so this is just to make
594 * sure. */
595 WaitForSingleObject( pxThreadState->pvYieldEvent, INFINITE );
596 }
597 }
598}
599/*-----------------------------------------------------------*/
600
601void vPortGenerateSimulatedInterruptFromWindowsThread( uint32_t ulInterruptNumber )
602{
603 if( xPortRunning == pdTRUE )
604 {
605 /* Can't proceed if in a critical section as pvInterruptEventMutex won't
606 * be available. */
607 WaitForSingleObject( pvInterruptEventMutex, INFINITE );
608
609 /* Pending a user defined interrupt to be handled in simulated interrupt
610 * handler thread. */
611 ulPendingInterrupts |= ( 1 << ulInterruptNumber );
612
613 /* The interrupt is now pending - notify the simulated interrupt
614 * handler thread. Must be outside of a critical section to get here so
615 * the handler thread can execute immediately pvInterruptEventMutex is
616 * released. */
617 configASSERT( ulCriticalNesting == 0UL );
618 SetEvent( pvInterruptEvent );
619
620 /* Give back the mutex so the simulated interrupt handler unblocks
621 * and can access the interrupt handler variables. */
622 ReleaseMutex( pvInterruptEventMutex );
623 }
624}

627void vPortSetInterruptHandler( uint32_t ulInterruptNumber,
628 uint32_t ( * pvHandler )( void ) )
629{
630 if( ulInterruptNumber < portMAX_INTERRUPTS )
631 {
632 if( pvInterruptEventMutex != NULL )
633 {
634 WaitForSingleObject( pvInterruptEventMutex, INFINITE );
635 ulIsrHandler[ ulInterruptNumber ] = pvHandler;
636 ReleaseMutex( pvInterruptEventMutex );
637 }
638 else
639 {
640 ulIsrHandler[ ulInterruptNumber ] = pvHandler;
641 }
642 }

644/*-----------------------------------------------------------*/
645
646void vPortEnterCritical( void )
647{
648 if( xPortRunning == pdTRUE )
649 {
650 /* The interrupt event mutex is held for the entire critical section,
651 * effectively disabling (simulated) interrupts. */
652 WaitForSingleObject( pvInterruptEventMutex, INFINITE );
653 }
654
655 ulCriticalNesting++;
656}
657/*-----------------------------------------------------------*/
658
659void vPortExitCritical( void )
660{
661 int32_t lMutexNeedsReleasing;
662
663 /* The interrupt event mutex should already be held by this thread as it was
664 * obtained on entry to the critical section. */
665 lMutexNeedsReleasing = pdTRUE;
666
667 if( ulCriticalNesting > portNO_CRITICAL_NESTING )
668 {
669 ulCriticalNesting--;
670
671 /* Don't need to wait for any pending interrupts to execute if the
672 * critical section was exited from inside an interrupt. */
673 if( ( ulCriticalNesting == portNO_CRITICAL_NESTING ) && ( xInsideInterrupt == pdFALSE ) )
674 {
675 /* Were any interrupts set to pending while interrupts were
676 * (simulated) disabled? */
677 if( ulPendingInterrupts != 0UL )
678 {
679 ThreadState_t * pxThreadState = ( ThreadState_t * ) *( ( size_t * ) pxCurrentTCB );
680
681 configASSERT( xPortRunning );
682
683 /* The interrupt won't actually executed until
684 * pvInterruptEventMutex is released as it waits on both
685 * pvInterruptEventMutex and pvInterruptEvent.
686 * pvInterruptEvent is only set when the simulated
687 * interrupt is pended if the interrupt is pended
688 * from outside a critical section - hence it is set
689 * here. */
690 SetEvent( pvInterruptEvent );
691
692 /* The calling task is going to wait for an event to ensure the
693 * interrupt that is pending executes immediately after the
694 * critical section is exited - so make sure the event is not
695 * already signaled. */
696 ResetEvent( pxThreadState->pvYieldEvent );
697
698 /* Mutex will be released now so the (simulated) interrupt can

29/*-----------------------------------------------------------
30* Implementation of functions defined in portable.h for the Posix port.
31*
32* Each task has a pthread which eases use of standard debuggers
33* (allowing backtraces of tasks etc). Threads for tasks that are not
34* running are blocked in sigwait().
35*
36* Task switch is done by resuming the thread for the next task by
37* signaling the condition variable and then waiting on a condition variable
38* with the current thread.
39*
40* The timer interrupt uses SIGALRM and care is taken to ensure that
41* the signal handler runs only on the thread for the current task.
42*
43* Use of part of the standard C library requires care as some
44* functions can take pthread mutexes internally which can result in
45* deadlocks as the FreeRTOS kernel can switch tasks while they're
46* holding a pthread mutex.
47*
48* stdio (printf() and friends) should be called from a single task
49* only or serialized with a FreeRTOS primitive such as a binary
50* semaphore or mutex.
51*----------------------------------------------------------*/

148 * See header file for description.
149 */
150StackType_t * pxPortInitialiseStack( StackType_t * pxTopOfStack,
151 StackType_t * pxEndOfStack,
152 TaskFunction_t pxCode,
153 void * pvParameters )
154{
155 Thread_t * thread;
156 pthread_attr_t xThreadAttributes;
157 size_t ulStackSize;
158 int iRet;
159
160 ( void ) pthread_once( &hSigSetupThread, prvSetupSignalsAndSchedulerPolicy );
161
162 /*
163 * Store the additional thread data at the start of the stack.
164 */
165 thread = ( Thread_t * ) ( pxTopOfStack + 1 ) - 1;
166 pxTopOfStack = ( StackType_t * ) thread - 1;
167
168 #ifdef __APPLE__
169 pxEndOfStack = ( StackType_t * ) mach_vm_round_page( pxEndOfStack );
170 #endif
171
172 ulStackSize = ( size_t ) ( pxTopOfStack + 1 - pxEndOfStack ) * sizeof( *pxTopOfStack );
173
174 #ifdef __APPLE__
175 ulStackSize = mach_vm_trunc_page( ulStackSize );
176 #endif
177
178 thread->pxCode = pxCode;
179 thread->pvParams = pvParameters;
180 thread->xDying = pdFALSE;
181
182 /* Ensure ulStackSize is at least PTHREAD_STACK_MIN */
183 ulStackSize = (ulStackSize < PTHREAD_STACK_MIN) ? PTHREAD_STACK_MIN : ulStackSize;
184
185 pthread_attr_init( &xThreadAttributes );
186 iRet = pthread_attr_setstacksize( &xThreadAttributes, ulStackSize );
187
188 if( iRet != 0 )
189 {
190 fprintf( stderr, "[WARN] pthread_attr_setstacksize failed with return value: %d. Default stack size will be used.\n", iRet );
191 }
192
193 thread->ev = event_create();
194
195 vPortEnterCritical();
196
197 iRet = pthread_create( &thread->pthread, &xThreadAttributes,
198 prvWaitForStart, thread );
199
200 if( iRet != 0 )
201 {
202 prvFatalError( "pthread_create", iRet );
203 }
204
205 vPortExitCritical();
206
207 return pxTopOfStack;
208}

220/*
221 * See header file for description.
222 */
223BaseType_t xPortStartScheduler( void )
224{
225 int iSignal;
226 sigset_t xSignals;
227
228 hMainThread = pthread_self();
229 prvPortSetCurrentThreadName("Scheduler");
230
231 /* Start the timer that generates the tick ISR(SIGALRM).
232 * Interrupts are disabled here already. */
233 prvSetupTimerInterrupt();
234
235 /*
236 * Block SIG_RESUME before starting any tasks so the main thread can sigwait on it.
237 * To sigwait on an unblocked signal is undefined.
238 * https://pubs.opengroup.org/onlinepubs/009604499/functions/sigwait.html
239 */
240 sigemptyset( &xSignals );
241 sigaddset( &xSignals, SIG_RESUME );
242 ( void ) pthread_sigmask( SIG_BLOCK, &xSignals, NULL );
243
244 /* Start the first task. */
245 vPortStartFirstTask();
246
247 /* Wait until signaled by vPortEndScheduler(). */
248 while( xSchedulerEnd != pdTRUE )
249 {
250 sigwait( &xSignals, &iSignal );
251 }
252
253 /*
254 * clear out the variable that is used to end the scheduler, otherwise
255 * subsequent scheduler restarts will end immediately.
256 */
257 xSchedulerEnd = pdFALSE;
258
259 /* Reset pthread_once_t, needed to restart the scheduler again.
260 * memset the internal struct members for MacOS/Linux Compatability */
261 #if __APPLE__
262 hSigSetupThread.__sig = _PTHREAD_ONCE_SIG_init;
263 memset( ( void * ) &hSigSetupThread.__opaque, 0, sizeof(hSigSetupThread.__opaque));
264 #else /* Linux PTHREAD library*/
265 hSigSetupThread = PTHREAD_ONCE_INIT;
266 #endif /* __APPLE__*/
267
268 /* Restore original signal mask. */
269 ( void ) pthread_sigmask( SIG_SETMASK, &xSchedulerOriginalSignalMask, NULL );
270
271 return 0;
272}

275void vPortEndScheduler( void )
276{
277 Thread_t * pxCurrentThread;
278
279 /* Stop the timer tick thread. */
280 xTimerTickThreadShouldRun = false;
281 pthread_join( hTimerTickThread, NULL );
282
283 /* Signal the scheduler to exit its loop. */
284 xSchedulerEnd = pdTRUE;
285 ( void ) pthread_kill( hMainThread, SIG_RESUME );
286
287 /* Waiting to be deleted here. */
288 pxCurrentThread = prvGetThreadFromTask( xTaskGetCurrentTaskHandle() );
289 event_wait( pxCurrentThread->ev );
290 pthread_testcancel();
291}

378/* commented as part of the code below in vPortSystemTickHandler,
379 * to adjust timing according to full demo requirements */
380/* static uint64_t prvTickCount; */
381
382static void * prvTimerTickHandler( void * arg )
383{
384 ( void ) arg;
385
386 prvPortSetCurrentThreadName("Scheduler timer");
387
388 while( xTimerTickThreadShouldRun )
389 {
390 /*
391 * signal to the active task to cause tick handling or
392 * preemption (if enabled)
393 */
394 Thread_t * thread = prvGetThreadFromTask( xTaskGetCurrentTaskHandle() );
395 pthread_kill( thread->pthread, SIGALRM );
396 usleep( portTICK_RATE_MICROSECONDS );
397 }
398
399 return NULL;
400}

416static void vPortSystemTickHandler( int sig )
417{
418 Thread_t * pxThreadToSuspend;
419 Thread_t * pxThreadToResume;
420
421 ( void ) sig;
422
423/* uint64_t xExpectedTicks; */
424
425 uxCriticalNesting++; /* Signals are blocked in this signal handler. */
426
427 #if ( configUSE_PREEMPTION == 1 )
428 pxThreadToSuspend = prvGetThreadFromTask( xTaskGetCurrentTaskHandle() );
429 #endif
430
431 /* Tick Increment, accounting for any lost signals or drift in
432 * the timer. */
433
434/*
435 * Comment code to adjust timing according to full demo requirements
436 * xExpectedTicks = (prvGetTimeNs() - prvStartTimeNs)
437 * / (portTICK_RATE_MICROSECONDS * 1000);
438 * do { */
439 xTaskIncrementTick();
440
441/* prvTickCount++;
442 * } while (prvTickCount < xExpectedTicks);
443 */
444
445 #if ( configUSE_PREEMPTION == 1 )
446 /* Select Next Task. */
447 vTaskSwitchContext();
448
449 pxThreadToResume = prvGetThreadFromTask( xTaskGetCurrentTaskHandle() );
450
451 prvSwitchThread( pxThreadToResume, pxThreadToSuspend );
452 #endif
453
454 uxCriticalNesting--;
455}

483static void * prvWaitForStart( void * pvParams )
484{
485 Thread_t * pxThread = pvParams;
486
487 prvSuspendSelf( pxThread );
488
489 /* Resumed for the first time, unblocks all signals. */
490 uxCriticalNesting = 0;
491 vPortEnableInterrupts();
492
493 /* Set thread name */
494 prvPortSetCurrentThreadName(pcTaskGetName(xTaskGetCurrentTaskHandle()));
495
496 /* Call the task's entry point. */
497 pxThread->pxCode( pxThread->pvParams );
498
499 /* A function that implements a task must not exit or attempt to return to
500 * its caller as there is nothing to return to. If a task wants to exit it
501 * should instead call vTaskDelete( NULL ). Artificially force an assert()
502 * to be triggered if configASSERT() is defined, so application writers can
503 * catch the error. */
504 configASSERT( pdFALSE );
505
506 return NULL;
507}

510static void prvSwitchThread( Thread_t * pxThreadToResume,
511 Thread_t * pxThreadToSuspend )
512{
513 BaseType_t uxSavedCriticalNesting;
514
515 if( pxThreadToSuspend != pxThreadToResume )
516 {
517 /*
518 * Switch tasks.
519 *
520 * The critical section nesting is per-task, so save it on the
521 * stack of the current (suspending thread), restoring it when
522 * we switch back to this task.
523 */
524 uxSavedCriticalNesting = uxCriticalNesting;
525
526 prvResumeThread( pxThreadToResume );
527
528 if( pxThreadToSuspend->xDying == pdTRUE )
529 {
530 pthread_exit( NULL );
531 }
532
533 prvSuspendSelf( pxThreadToSuspend );
534
535 uxCriticalNesting = uxSavedCriticalNesting;
536 }
537}

540static void prvSuspendSelf( Thread_t * thread )
541{
542 /*
543 * Suspend this thread by waiting for a pthread_cond_signal event.
544 *
545 * A suspended thread must not handle signals (interrupts) so
546 * all signals must be blocked by calling this from:
547 *
548 * - Inside a critical section (vPortEnterCritical() /
549 * vPortExitCritical()).
550 *
551 * - From a signal handler that has all signals masked.
552 *
553 * - A thread with all signals blocked with pthread_sigmask().
554 */
555 event_wait( thread->ev );
556 pthread_testcancel();
557}

570static void prvSetupSignalsAndSchedulerPolicy( void )
571{
572 struct sigaction sigtick;
573 int iRet;
574
575 hMainThread = pthread_self();
576
577 /* Initialise common signal masks. */
578 sigfillset( &xAllSignals );
579
580 /* Don't block SIGINT so this can be used to break into GDB while
581 * in a critical section. */
582 sigdelset( &xAllSignals, SIGINT );
583
584 /*
585 * Block all signals in this thread so all new threads
586 * inherits this mask.
587 *
588 * When a thread is resumed for the first time, all signals
589 * will be unblocked.
590 */
591 ( void ) pthread_sigmask( SIG_SETMASK,
592 &xAllSignals,
593 &xSchedulerOriginalSignalMask );
594
595 sigtick.sa_flags = 0;
596 sigtick.sa_handler = vPortSystemTickHandler;
597 sigfillset( &sigtick.sa_mask );
598
599 iRet = sigaction( SIGALRM, &sigtick, NULL );
600
601 if( iRet == -1 )
602 {
603 prvFatalError( "sigaction", errno );
604 }
605}

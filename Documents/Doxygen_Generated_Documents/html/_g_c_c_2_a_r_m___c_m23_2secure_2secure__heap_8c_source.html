72/* MSB of the xBlockSize member of an BlockLink_t structure is used to track
73 * the allocation status of a block. When MSB of the xBlockSize member of
74 * an BlockLink_t structure is set then the block belongs to the application.
75 * When the bit is free the block is still part of the free heap space. */
76#define secureheapBLOCK_ALLOCATED_BITMASK ( ( ( size_t ) 1 ) << ( ( sizeof( size_t ) * secureheapBITS_PER_BYTE ) - 1 ) )
77#define secureheapBLOCK_SIZE_IS_VALID( xBlockSize ) ( ( ( xBlockSize ) & secureheapBLOCK_ALLOCATED_BITMASK ) == 0 )
78#define secureheapBLOCK_IS_ALLOCATED( pxBlock ) ( ( ( pxBlock->xBlockSize ) & secureheapBLOCK_ALLOCATED_BITMASK ) != 0 )
79#define secureheapALLOCATE_BLOCK( pxBlock ) ( ( pxBlock->xBlockSize ) |= secureheapBLOCK_ALLOCATED_BITMASK )
80#define secureheapFREE_BLOCK( pxBlock ) ( ( pxBlock->xBlockSize ) &= ~secureheapBLOCK_ALLOCATED_BITMASK )

144static void prvHeapInit( void )
145{
146 BlockLink_t * pxFirstFreeBlock;
147 uint8_t * pucAlignedHeap;
148 size_t uxAddress;
149 size_t xTotalHeapSize = secureconfigTOTAL_HEAP_SIZE;
150
151 /* Ensure the heap starts on a correctly aligned boundary. */
152 uxAddress = ( size_t ) ucHeap;
153
154 if( ( uxAddress & secureportBYTE_ALIGNMENT_MASK ) != 0 )
155 {
156 uxAddress += ( secureportBYTE_ALIGNMENT - 1 );
157 uxAddress &= ~( ( size_t ) secureportBYTE_ALIGNMENT_MASK );
158 xTotalHeapSize -= uxAddress - ( size_t ) ucHeap;
159 }
160
161 pucAlignedHeap = ( uint8_t * ) uxAddress;
162
163 /* xStart is used to hold a pointer to the first item in the list of free
164 * blocks. The void cast is used to prevent compiler warnings. */
165 xStart.pxNextFreeBlock = ( void * ) pucAlignedHeap;
166 xStart.xBlockSize = ( size_t ) 0;
167
168 /* pxEnd is used to mark the end of the list of free blocks and is inserted
169 * at the end of the heap space. */
170 uxAddress = ( ( size_t ) pucAlignedHeap ) + xTotalHeapSize;
171 uxAddress -= xHeapStructSize;
172 uxAddress &= ~( ( size_t ) secureportBYTE_ALIGNMENT_MASK );
173 pxEnd = ( void * ) uxAddress;
174 pxEnd->xBlockSize = 0;
175 pxEnd->pxNextFreeBlock = NULL;
176
177 /* To start with there is a single free block that is sized to take up the
178 * entire heap space, minus the space taken by pxEnd. */
179 pxFirstFreeBlock = ( void * ) pucAlignedHeap;
180 pxFirstFreeBlock->xBlockSize = uxAddress - ( size_t ) pxFirstFreeBlock;
181 pxFirstFreeBlock->pxNextFreeBlock = pxEnd;
182
183 /* Only one block exists - and it covers the entire usable heap space. */
184 xMinimumEverFreeBytesRemaining = pxFirstFreeBlock->xBlockSize;
185 xFreeBytesRemaining = pxFirstFreeBlock->xBlockSize;
186}

189static void prvInsertBlockIntoFreeList( BlockLink_t * pxBlockToInsert )
190{
191 BlockLink_t * pxIterator;
192 uint8_t * puc;
193
194 /* Iterate through the list until a block is found that has a higher address
195 * than the block being inserted. */
196 for( pxIterator = &xStart; pxIterator->pxNextFreeBlock < pxBlockToInsert; pxIterator = pxIterator->pxNextFreeBlock )
197 {
198 /* Nothing to do here, just iterate to the right position. */
199 }
200
201 /* Do the block being inserted, and the block it is being inserted after
202 * make a contiguous block of memory? */
203 puc = ( uint8_t * ) pxIterator;
204
205 if( ( puc + pxIterator->xBlockSize ) == ( uint8_t * ) pxBlockToInsert )
206 {
207 pxIterator->xBlockSize += pxBlockToInsert->xBlockSize;
208 pxBlockToInsert = pxIterator;
209 }
210 else
211 {
212 mtCOVERAGE_TEST_MARKER();
213 }
214
215 /* Do the block being inserted, and the block it is being inserted before
216 * make a contiguous block of memory? */
217 puc = ( uint8_t * ) pxBlockToInsert;
218
219 if( ( puc + pxBlockToInsert->xBlockSize ) == ( uint8_t * ) pxIterator->pxNextFreeBlock )
220 {
221 if( pxIterator->pxNextFreeBlock != pxEnd )
222 {
223 /* Form one big block from the two blocks. */
224 pxBlockToInsert->xBlockSize += pxIterator->pxNextFreeBlock->xBlockSize;
225 pxBlockToInsert->pxNextFreeBlock = pxIterator->pxNextFreeBlock->pxNextFreeBlock;
226 }
227 else
228 {
229 pxBlockToInsert->pxNextFreeBlock = pxEnd;
230 }
231 }
232 else
233 {
234 pxBlockToInsert->pxNextFreeBlock = pxIterator->pxNextFreeBlock;
235 }
236
237 /* If the block being inserted plugged a gab, so was merged with the block
238 * before and the block after, then it's pxNextFreeBlock pointer will have
239 * already been set, and should not be set here as that would make it point
240 * to itself. */
241 if( pxIterator != pxBlockToInsert )
242 {
243 pxIterator->pxNextFreeBlock = pxBlockToInsert;
244 }
245 else
246 {
247 mtCOVERAGE_TEST_MARKER();
248 }
249}

252void * pvPortMalloc( size_t xWantedSize )
253{
254 BlockLink_t * pxBlock;
255 BlockLink_t * pxPreviousBlock;
256 BlockLink_t * pxNewBlockLink;
257 void * pvReturn = NULL;
258 size_t xAdditionalRequiredSize;
259 size_t xAllocatedBlockSize = 0;
260
261 /* If this is the first call to malloc then the heap will require
262 * initialisation to setup the list of free blocks. */
263 if( pxEnd == NULL )
264 {
265 prvHeapInit();
266 }
267 else
268 {
269 mtCOVERAGE_TEST_MARKER();
270 }
271
272 if( xWantedSize > 0 )
273 {
274 /* The wanted size must be increased so it can contain a BlockLink_t
275 * structure in addition to the requested amount of bytes. */
276 if( secureheapADD_WILL_OVERFLOW( xWantedSize, xHeapStructSize ) == 0 )
277 {
278 xWantedSize += xHeapStructSize;
279
280 /* Ensure that blocks are always aligned to the required number
281 * of bytes. */
282 if( ( xWantedSize & secureportBYTE_ALIGNMENT_MASK ) != 0x00 )
283 {
284 /* Byte alignment required. */
285 xAdditionalRequiredSize = secureportBYTE_ALIGNMENT - ( xWantedSize & secureportBYTE_ALIGNMENT_MASK );
286
287 if( secureheapADD_WILL_OVERFLOW( xWantedSize, xAdditionalRequiredSize ) == 0 )
288 {
289 xWantedSize += xAdditionalRequiredSize;
290 }
291 else
292 {
293 xWantedSize = 0;
294 }
295 }
296 else
297 {
298 mtCOVERAGE_TEST_MARKER();
299 }
300 }
301 else
302 {
303 xWantedSize = 0;
304 }
305 }
306 else
307 {
308 mtCOVERAGE_TEST_MARKER();
309 }
310
311 /* Check the requested block size is not so large that the top bit is set.
312 * The top bit of the block size member of the BlockLink_t structure is used
313 * to determine who owns the block - the application or the kernel, so it
314 * must be free. */
315 if( secureheapBLOCK_SIZE_IS_VALID( xWantedSize ) != 0 )
316 {
317 if( ( xWantedSize > 0 ) && ( xWantedSize <= xFreeBytesRemaining ) )
318 {
319 /* Traverse the list from the start (lowest address) block until
320 * one of adequate size is found. */
321 pxPreviousBlock = &xStart;
322 pxBlock = xStart.pxNextFreeBlock;
323
324 while( ( pxBlock->xBlockSize < xWantedSize ) && ( pxBlock->pxNextFreeBlock != NULL ) )
325 {
326 pxPreviousBlock = pxBlock;
327 pxBlock = pxBlock->pxNextFreeBlock;
328 }
329
330 /* If the end marker was reached then a block of adequate size was
331 * not found. */
332 if( pxBlock != pxEnd )
333 {
334 /* Return the memory space pointed to - jumping over the
335 * BlockLink_t structure at its start. */
336 pvReturn = ( void * ) ( ( ( uint8_t * ) pxPreviousBlock->pxNextFreeBlock ) + xHeapStructSize );
337
338 /* This block is being returned for use so must be taken out
339 * of the list of free blocks. */
340 pxPreviousBlock->pxNextFreeBlock = pxBlock->pxNextFreeBlock;
341
342 /* If the block is larger than required it can be split into
343 * two. */
344 if( ( pxBlock->xBlockSize - xWantedSize ) > secureheapMINIMUM_BLOCK_SIZE )
345 {
346 /* This block is to be split into two. Create a new
347 * block following the number of bytes requested. The void
348 * cast is used to prevent byte alignment warnings from the
349 * compiler. */
350 pxNewBlockLink = ( void * ) ( ( ( uint8_t * ) pxBlock ) + xWantedSize );
351 secureportASSERT( ( ( ( size_t ) pxNewBlockLink ) & secureportBYTE_ALIGNMENT_MASK ) == 0 );
352
353 /* Calculate the sizes of two blocks split from the single
354 * block. */
355 pxNewBlockLink->xBlockSize = pxBlock->xBlockSize - xWantedSize;
356 pxBlock->xBlockSize = xWantedSize;
357
358 /* Insert the new block into the list of free blocks. */
359 pxNewBlockLink->pxNextFreeBlock = pxPreviousBlock->pxNextFreeBlock;
360 pxPreviousBlock->pxNextFreeBlock = pxNewBlockLink;
361 }
362 else
363 {
364 mtCOVERAGE_TEST_MARKER();
365 }
366
367 xFreeBytesRemaining -= pxBlock->xBlockSize;
368
369 if( xFreeBytesRemaining < xMinimumEverFreeBytesRemaining )
370 {
371 xMinimumEverFreeBytesRemaining = xFreeBytesRemaining;
372 }
373 else
374 {
375 mtCOVERAGE_TEST_MARKER();
376 }
377
378 xAllocatedBlockSize = pxBlock->xBlockSize;
379
380 /* The block is being returned - it is allocated and owned by
381 * the application and has no "next" block. */
382 secureheapALLOCATE_BLOCK( pxBlock );
383 pxBlock->pxNextFreeBlock = NULL;
384 }
385 else
386 {
387 mtCOVERAGE_TEST_MARKER();
388 }
389 }
390 else
391 {
392 mtCOVERAGE_TEST_MARKER();
393 }
394 }
395 else
396 {
397 mtCOVERAGE_TEST_MARKER();
398 }
399
400 traceMALLOC( pvReturn, xAllocatedBlockSize );
401
402 /* Prevent compiler warnings when trace macros are not used. */
403 ( void ) xAllocatedBlockSize;
404
405 #if ( secureconfigUSE_MALLOC_FAILED_HOOK == 1 )
406 {
407 if( pvReturn == NULL )
408 {
409 extern void vApplicationMallocFailedHook( void );
410 vApplicationMallocFailedHook();
411 }
412 else
413 {
414 mtCOVERAGE_TEST_MARKER();
415 }
416 }
417 #endif /* if ( secureconfigUSE_MALLOC_FAILED_HOOK == 1 ) */
418
419 secureportASSERT( ( ( ( size_t ) pvReturn ) & ( size_t ) secureportBYTE_ALIGNMENT_MASK ) == 0 );
420 return pvReturn;
421}

424void vPortFree( void * pv )
425{
426 uint8_t * puc = ( uint8_t * ) pv;
427 BlockLink_t * pxLink;
428
429 if( pv != NULL )
430 {
431 /* The memory being freed will have an BlockLink_t structure immediately
432 * before it. */
433 puc -= xHeapStructSize;
434
435 /* This casting is to keep the compiler from issuing warnings. */
436 pxLink = ( void * ) puc;
437
438 /* Check the block is actually allocated. */
439 secureportASSERT( secureheapBLOCK_IS_ALLOCATED( pxLink ) != 0 );
440 secureportASSERT( pxLink->pxNextFreeBlock == NULL );
441
442 if( secureheapBLOCK_IS_ALLOCATED( pxLink ) != 0 )
443 {
444 if( pxLink->pxNextFreeBlock == NULL )
445 {
446 /* The block is being returned to the heap - it is no longer
447 * allocated. */
448 secureheapFREE_BLOCK( pxLink );
449
450 secureportDISABLE_NON_SECURE_INTERRUPTS();
451 {
452 /* Add this block to the list of free blocks. */
453 xFreeBytesRemaining += pxLink->xBlockSize;
454 traceFREE( pv, pxLink->xBlockSize );
455 prvInsertBlockIntoFreeList( ( ( BlockLink_t * ) pxLink ) );
456 }
457 secureportENABLE_NON_SECURE_INTERRUPTS();
458 }
459 else
460 {
461 mtCOVERAGE_TEST_MARKER();
462 }
463 }
464 else
465 {
466 mtCOVERAGE_TEST_MARKER();
467 }
468 }
469}

137// When coprocessors are defined, we to maintain a pointer to coprocessors area.
138// We currently use a hack: redefine field xMPU_SETTINGS in TCB block as a structure that can hold:
139// MPU wrappers, coprocessor area pointer, trace code structure, and more if needed.
140// The field is normally used for memory protection. FreeRTOS should create another general purpose field.
141typedef struct {
142 #if XCHAL_CP_NUM > 0
143 volatile StackType_t* coproc_area; // Pointer to coprocessor save area; MUST BE FIRST
144 #endif
145
146 #if portUSING_MPU_WRAPPERS
147 // Define here mpu_settings, which is port dependent
148 int mpu_setting; // Just a dummy example here; MPU not ported to Xtensa yet
149 #endif
150
151 #if configUSE_TRACE_FACILITY_2
152 struct {
153 // Cf. porttraceStamp()
154 int taskstamp; /* Stamp from inside task to see where we are */
155 int taskstampcount; /* A counter usually incremented when we restart the task's loop */
156 } porttrace;
157 #endif
158} xMPU_SETTINGS;
159
160// Main hack to use MPU_wrappers even when no MPU is defined (warning: mpu_setting should not be accessed; otherwise move this above xMPU_SETTINGS)
161#if (XCHAL_CP_NUM > 0 || configUSE_TRACE_FACILITY_2) && !portUSING_MPU_WRAPPERS // If MPU wrappers not used, we still need to allocate coproc area
162 #undef portUSING_MPU_WRAPPERS
163 #define portUSING_MPU_WRAPPERS 1 // Enable it to allocate coproc area
164 #define MPU_WRAPPERS_H // Override mpu_wrapper.h to disable unwanted code
165 #define PRIVILEGED_FUNCTION
166 #define PRIVILEGED_DATA
167#endif

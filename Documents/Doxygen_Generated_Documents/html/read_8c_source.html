99size_t __read( int handle,
100 uint8_t * buffer,
101 size_t size )
102{
103 int nChars = 0;
104
105 /* This implementation only reads from stdin. */
106 /* For all other file handles, it returns failure. */
107 if( handle != _LLIO_STDIN )
108 {
109 return _LLIO_ERROR;
110 }
111
112 for( ; size > 0; --size )
113 {
114 int c = usart_getchar( stdio_usart_base );
115
116 if( c < 0 )
117 {
118 break;
119 }
120
121 *buffer++ = c;
122 ++nChars;
123 }
124
125 return nChars;
126}

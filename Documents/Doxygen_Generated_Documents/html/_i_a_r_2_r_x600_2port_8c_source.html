73StackType_t * pxPortInitialiseStack( StackType_t * pxTopOfStack,
74 TaskFunction_t pxCode,
75 void * pvParameters )
76{
77 /* R0 is not included as it is the stack pointer. */
78
79 *pxTopOfStack = 0x00;
80 pxTopOfStack--;
81 *pxTopOfStack = portINITIAL_PSW;
82 pxTopOfStack--;
83 *pxTopOfStack = ( StackType_t ) pxCode;
84
85 /* When debugging it can be useful if every register is set to a known
86 * value. Otherwise code space can be saved by just setting the registers
87 * that need to be set. */
88 #ifdef USE_FULL_REGISTER_INITIALISATION
89 {
90 pxTopOfStack--;
91 *pxTopOfStack = 0xffffffff; /* r15. */
92 pxTopOfStack--;
93 *pxTopOfStack = 0xeeeeeeee;
94 pxTopOfStack--;
95 *pxTopOfStack = 0xdddddddd;
96 pxTopOfStack--;
97 *pxTopOfStack = 0xcccccccc;
98 pxTopOfStack--;
99 *pxTopOfStack = 0xbbbbbbbb;
100 pxTopOfStack--;
101 *pxTopOfStack = 0xaaaaaaaa;
102 pxTopOfStack--;
103 *pxTopOfStack = 0x99999999;
104 pxTopOfStack--;
105 *pxTopOfStack = 0x88888888;
106 pxTopOfStack--;
107 *pxTopOfStack = 0x77777777;
108 pxTopOfStack--;
109 *pxTopOfStack = 0x66666666;
110 pxTopOfStack--;
111 *pxTopOfStack = 0x55555555;
112 pxTopOfStack--;
113 *pxTopOfStack = 0x44444444;
114 pxTopOfStack--;
115 *pxTopOfStack = 0x33333333;
116 pxTopOfStack--;
117 *pxTopOfStack = 0x22222222;
118 pxTopOfStack--;
119 }
120 #else /* ifdef USE_FULL_REGISTER_INITIALISATION */
121 {
122 pxTopOfStack -= 15;
123 }
124 #endif /* ifdef USE_FULL_REGISTER_INITIALISATION */
125
126 *pxTopOfStack = ( StackType_t ) pvParameters; /* R1 */
127 pxTopOfStack--;
128 *pxTopOfStack = portINITIAL_FPSW;
129 pxTopOfStack--;
130 *pxTopOfStack = 0x12345678; /* Accumulator. */
131 pxTopOfStack--;
132 *pxTopOfStack = 0x87654321; /* Accumulator. */
133
134 return pxTopOfStack;
135}

138BaseType_t xPortStartScheduler( void )
139{
140 extern void vApplicationSetupTimerInterrupt( void );
141
142 /* Use pxCurrentTCB just so it does not get optimised away. */
143 if( pxCurrentTCB != NULL )
144 {
145 /* Call an application function to set up the timer that will generate the
146 * tick interrupt. This way the application can decide which peripheral to
147 * use. A demo application is provided to show a suitable example. */
148 vApplicationSetupTimerInterrupt();
149
150 /* Enable the software interrupt. */
151 _IEN( _ICU_SWINT ) = 1;
152
153 /* Ensure the software interrupt is clear. */
154 _IR( _ICU_SWINT ) = 0;
155
156 /* Ensure the software interrupt is set to the kernel priority. */
157 _IPR( _ICU_SWINT ) = configKERNEL_INTERRUPT_PRIORITY;
158
159 /* Start the first task. */
160 prvStartFirstTask();
161 }
162
163 /* Should not get here. */
164 return pdFAIL;
165}

168#pragma vector = configTICK_VECTOR
169__interrupt void vTickISR( void )
170{
171 /* Re-enable interrupts. */
172 __enable_interrupt();
173
174 /* Increment the tick, and perform any processing the new tick value
175 * necessitates. */
176 __set_interrupt_level( configMAX_SYSCALL_INTERRUPT_PRIORITY );
177 {
178 if( xTaskIncrementTick() != pdFALSE )
179 {
180 taskYIELD();
181 }
182 }
183 __set_interrupt_level( configKERNEL_INTERRUPT_PRIORITY );
184}

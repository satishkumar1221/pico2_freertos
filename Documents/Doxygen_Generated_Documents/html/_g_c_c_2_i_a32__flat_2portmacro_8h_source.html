111/* Don't yield if inside a critical section - instead hold the yield pending
112 * so it is performed when the critical section is exited. */
113#define portYIELD() \
114 { \
115 extern volatile uint32_t ulCriticalNesting; \
116 extern volatile uint32_t ulPortYieldPending; \
117 if( ulCriticalNesting != 0 ) \
118 { \
119 ulPortYieldPending = pdTRUE; \
120 } \
121 else \
122 { \
123 __asm volatile ( portYIELD_INTERRUPT ); \
124 } \
125 }
126
127/* Called at the end of an ISR that can cause a context switch - pend a yield if
128 * xSwitchRequired is not false. */
129#define portEND_SWITCHING_ISR( xSwitchRequired ) \
130 { \
131 extern volatile uint32_t ulPortYieldPending; \
132 if( xSwitchRequired != pdFALSE ) \
133 { \
134 ulPortYieldPending = 1; \
135 } \
136 }

138/* Same as portEND_SWITCHING_ISR() - take your pick which name to use. */

219/* The FreeRTOS scheduling algorithm selects the task that will enter the
220 * Running state. configUSE_PORT_OPTIMISED_TASK_SELECTION is used to set how
221 * that is done.
222 *
223 * If configUSE_PORT_OPTIMISED_TASK_SELECTION is set to 0 then the task to
224 * enter the Running state is selected using a portable algorithm written in
225 * C. This is the slowest method, but the algorithm does not restrict the
226 * maximum number of unique RTOS task priorities that are available.
227 *
228 * If configUSE_PORT_OPTIMISED_TASK_SELECTION is set to 1 then the task to
229 * enter the Running state is selected using a single assembly instruction.
230 * This is the fastest method, but restricts the maximum number of unique RTOS
231 * task priorities to 32 (the same task priority can be assigned to any number
232 * of RTOS tasks). */
233 #warning configUSE_PORT_OPTIMISED_TASK_SELECTION was not defined in FreeRTOSConfig.h and has been defaulted to 1
234 #define configUSE_PORT_OPTIMISED_TASK_SELECTION 1
235#endif

237#ifndef configUSE_COMMON_INTERRUPT_ENTRY_POINT
238
239/* There are two ways of implementing interrupt handlers:
240 *
241 * 1) As standard C functions -
242 *
243 * This method can only be used if configUSE_COMMON_INTERRUPT_ENTRY_POINT
244 * is set to 1. The C function is installed using
245 * xPortRegisterCInterruptHandler().
246 *
247 * This is the simplest of the two methods but incurs a slightly longer
248 * interrupt entry time.
249 *
250 * 2) By using an assembly stub that wraps the handler in the FreeRTOS
251 * portFREERTOS_INTERRUPT_ENTRY and portFREERTOS_INTERRUPT_EXIT macros.
252 *
253 * This method can always be used. It is slightly more complex than
254 * method 1 but benefits from a faster interrupt entry time. */
255 #warning configUSE_COMMON_INTERRUPT_ENTRY_POINT was not defined in FreeRTOSConfig.h and has been defaulted to 1.
256 #define configUSE_COMMON_INTERRUPT_ENTRY_POINT 1
257#endif
